## Goal

Read-only endpoint that returns the current status and all fields for a given R2P request. Available to both originating and receiving participants. Enables demo observability — callers can confirm routing, acknowledgement, and payment state transitions without needing a database.

## API Contract

```
GET /r2p/requests/{r2pId}
  Response 200: {
    r2pId:           string   // UUID v7
    status:          string   // current R2PStatus
    payerId:         string
    payeeId:         string
    amount:          number
    currency:        string
    dueDate:         string   // YYYY-MM-DD
    expiryTimestamp: string   // ISO 8601
    remittanceInfo:  string | null
    createdAt:       string   // ISO 8601
    updatedAt:       string   // ISO 8601
  }
  Errors:
```

## Service: getRequest()

```typescript
// src/services/r2pRequest.ts

export type GetRequestResult =
  | { ok: true;  request: R2PRequestResponse }
  | { ok: false; code: 'NOT_FOUND'; message: string }

export interface R2PRequestResponse {
  r2pId:           string
  status:          string
  payerId:         string
  payeeId:         string
  amount:          number
  currency:        string
  dueDate:         string
  expiryTimestamp: string
  remittanceInfo:  string | null
  createdAt:       string
  updatedAt:       string
}

export function getRequest(r2pId: string): GetRequestResult
```

### Execution steps

1. **Fetch** — `r2pRepo.findById(r2pId)`
   - If not found: return `{ ok: false, code: 'NOT_FOUND', message: 'R2P request not found' }`

2. **Map** — translate `R2PRequestRow` to `R2PRequestResponse`:
   ```typescript
   {
     r2pId:           row.id,
     status:          row.status,
     payerId:         row.payer_id,
     payeeId:         row.payee_id,
     amount:          row.amount,
     currency:        row.currency,
     dueDate:         row.due_date,
     expiryTimestamp: row.expiry_timestamp,
     remittanceInfo:  row.remittance_info,
     createdAt:       row.created_at,
     updatedAt:       row.updated_at,
   }
   ```

3. **Return** — `{ ok: true, request }`

## Route Handler

```typescript
// src/routes/r2pRequests.ts

const getRequestHandler: RequestHandler = (req, res, _next) => {
  const { r2pId } = req.params
  const result = getRequest(r2pId)
  if (!result.ok) {
    res.status(404).json({ code: result.code, message: result.message })
    return
  }
  res.status(200).json(result.request)
}

// Mount:
router.get('/r2p/requests/:r2pId', getRequestHandler)
```

```typescript
it('returns full request fields for a known r2pId', () => {
```
